[c#]cefsharp的使用
cefsharp项目源码: https://github.com/cefsharp/CefSharp
首先引用cefsharp: Nugget 中搜索并安装 CefSharp.Winform, 并引用命名空间CefSharp.WinForms和CefSharp

//配置cefsharp初始设置

var setting = new CefSettings();

setting.Locale = "zh-CN"; //缓存路径

setting.CachePath = Application.StartupPath+"\\BrowserCache"; //浏览器引擎的语言

setting.AcceptLanguageList = "zh-CN,zh;q=0.8";

setting.LocalesDirPath = Application.StartupPath + "\\locales"; //日志文件

setting.LogFile = Application.StartupPath + "\\LogData";

setting.PersistSessionCookies = true;

 

//支持flash播放(需要在指定目录下放置flash pp版dll)

setting.UserDataPath = Application.StartupPath + "\\userData"; setting.CefCommandLineArgs.Add("enable-npapi", "1");

setting.CefCommandLineArgs.Add("ppapi-flash-path", Environment.CurrentDirectory + "\\plugins\\pepflashplayer32_32_0_0_207.dll");

setting.CefCommandLineArgs.Add("ppapi-flash-version", "32.0.0.207");

 

//设置cef

Cef.Initialize(setting, true, browserProcessHandler: null);

 

 

 

//创建浏览器部分代码

webBrower = new ChromiumWebBrowser(url); //url为初始地址

webBrower.Dock = DockStyle.Fill;// 填充方式

Controls.Add(webBrower); //添加浏览器
